Comandos curl para Testing

# GET Request
curl http://localhost:3000/api/item/123

# POST Request  
curl -X POST http://localhost:3000/api/item \
  -H "Content-Type: application/json" \
  -d '{"name":"test","value":123}'

# PUT Request
curl -X PUT http://localhost:3000/api/item/123 \
  -H "Content-Type: application/json" \
  -d '{"updated":true}'

# DELETE Request
curl -X DELETE http://localhost:3000/api/item/123
